10  import com.github.searls.jasmine.config.JasmineConfiguration;
11  import com.github.searls.jasmine.io.scripts.BasicScriptResolver;
12  import com.github.searls.jasmine.io.scripts.ContextPathScriptResolver;
13  import com.github.searls.jasmine.io.scripts.ScriptResolver;
14  
15  public class CreatesRunner {
16  
17    private final JasmineConfiguration config;
18  
19    private final Log log;
20    private final String runnerFileName;
21    private final ReporterType reporterType;
22  
23    public CreatesRunner(JasmineConfiguration config, Log log, String runnerFileName, ReporterType reporterType) {
24      this.config = config;
25      this.runnerFileName = runnerFileName;
26      this.reporterType = reporterType;
27      this.log = log;
28    }
29  
30    public String getRunnerFile() {
31      return this.runnerFileName;
32    }
33  
34    public void create() throws IOException {
35      File runnerDestination = new File(this.config.getJasmineTargetDir(),this.runnerFileName);
36      ScriptResolver resolver = new BasicScriptResolver(
37          config.getBasedir(),
38          config.getSources(),
39          config.getSpecs(),
40          config.getPreloadSources());
41      resolver = new ContextPathScriptResolver(
42          resolver,
43          config.getSrcDirectoryName(),
44          config.getSpecDirectoryName());
45  
46      SpecRunnerHtmlGenerator generator = new SpecRunnerHtmlGeneratorFactory().create(this.reporterType, this.config, resolver);
47  
48      String newRunnerHtml = generator.generate();
49      if(this.newRunnerDiffersFromOldRunner(runnerDestination, newRunnerHtml)) {
50        this.saveRunner(runnerDestination, newRunnerHtml);
51      } else {
52        this.log.info("Skipping spec runner generation, because an identical spec runner already exists.");
53      }
54    }
55  
56    private String existingRunner(File destination) throws IOException {
57      String existingRunner = null;
58      try {
59        if(destination.exists()) {
60          existingRunner = FileUtils.readFileToString(destination);
61        }
62      } catch(IOException e) {
63        this.log.warn("An error occurred while trying to open an existing manual spec runner. Continuing.");
64      }
65      return existingRunner;
66    }
67  
68    private boolean newRunnerDiffersFromOldRunner(File runnerDestination, String newRunner) throws IOException {
69      return !StringUtils.equals(newRunner, this.existingRunner(runnerDestination));
70    }
71  
72    private void saveRunner(File runnerDestination, String newRunner) throws IOException {
73      FileUtils.writeStringToFile(runnerDestination, newRunner, this.config.getSourceEncoding());
74    }
75  }
