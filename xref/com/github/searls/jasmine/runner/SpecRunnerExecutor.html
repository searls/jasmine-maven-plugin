20  public class SpecRunnerExecutor {
21  
22  	public static final String BUILD_REPORT_JS = "/lib/buildReport.js";
23  	public static final String CREATE_JUNIT_XML = "/lib/createJunitXml.js";
24  
25  	private final IOUtilsWrapper ioUtilsWrapper = new IOUtilsWrapper();
26  
27  	public JasmineResult execute(URL runnerUrl, File junitXmlReport, WebDriver driver, int timeout, boolean debug, Log log, String format) {
28  		try {
29  			if (!(driver instanceof JavascriptExecutor)) {
30  				throw new RuntimeException("The provided web driver can't execute JavaScript: " + driver.getClass());
31  			}
32  			JavascriptExecutor executor = (JavascriptExecutor) driver;
33  			driver.get(runnerUrl.toString());
34  			this.waitForRunnerToFinish(driver, timeout, debug, log);
35  
36        this.checkForConsoleErrors(driver, log);
37  
38  			JasmineResult jasmineResult = new JasmineResult();
39  			jasmineResult.setDetails(this.buildReport(executor,format));
40  			FileUtils.writeStringToFile(junitXmlReport, this.buildJunitXmlReport(executor,debug), "UTF-8");
41  			driver.quit();
42  
43  			return jasmineResult;
44  		} catch (Exception e) {
45  			throw new RuntimeException(e);
46  		}
47  	}
48  
49    private void checkForConsoleErrors(WebDriver driver, Log log) {
50      WebElement head = driver.findElement(By.tagName("head"));
51      if (head != null) {
52        String jserrors = head.getAttribute("jmp_jserror");
53        if (StringUtils.isNotBlank(jserrors)) {
54          log.warn("JavaScript Console Errors:\n\n  * "+jserrors.replaceAll(":!:","\n  * ")+"\n\n");
55          throw new RuntimeException("There were javascript console errors.");
56        }
57      }
58    }
59  
60  	private String buildReport(JavascriptExecutor driver, String format) throws IOException {
61  		String script =
62  				this.ioUtilsWrapper.toString(BUILD_REPORT_JS) +
63  				"return jasmineMavenPlugin.printReport(window.reporter,{format:'"+format+"'});";
64  		Object report = driver.executeScript(script);
65  		return report.toString();
66  	}
67  
68  	private String buildJunitXmlReport(JavascriptExecutor driver, boolean debug) throws IOException {
69  		Object junitReport = driver.executeScript(
70  				this.ioUtilsWrapper.toString(CREATE_JUNIT_XML) +
71  				"return junitXmlReporter.report(window.reporter,"+debug+");");
72  		return junitReport.toString();
73  	}
74  
75  	private void waitForRunnerToFinish(final WebDriver driver, int timeout, boolean debug, Log log) throws InterruptedException {
76  		final JavascriptExecutor executor = (JavascriptExecutor) driver;
77  		new WebDriverWait(driver, timeout, 1000).until(new Predicate<WebDriver>() {
78  			@Override
79  			public boolean apply(WebDriver input) {
80  				return SpecRunnerExecutor.this.executionFinished(executor);
81  			}
82  		});
83  
84  		if (!this.executionFinished(executor)) {
85  			this.handleTimeout(timeout, debug, log);
86  		}
87  	}
88  
89  	private void handleTimeout(int timeout, boolean debug, Log log) {
90  		log.warn("Attempted to wait for your specs to finish processing over the course of " +
91  				timeout +
92  				" seconds, but it still appears to be running.");
93  		if(debug) {
94  			log.warn("Debug mode: will attempt to parse the incomplete spec runner results");
95  		} else {
96  			throw new IllegalStateException("Timeout occurred. Aborting execution of specs. (Try configuring 'debug' to 'true' for more details.)");
97  		}
98  	}
99  
100 	private Boolean executionFinished(JavascriptExecutor driver) {
101 		return (Boolean) driver.executeScript("return (window.reporter === undefined) ? false : window.reporter.finished");
102 	}
