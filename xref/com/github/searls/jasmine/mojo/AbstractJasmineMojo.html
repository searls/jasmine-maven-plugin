50  	 * before other sources (and specs) in a particular order. Each source will first be

387 	@Override
388 	public void execute() throws MojoExecutionException, MojoFailureException {
389 
390 		this.loadResources();
391 
392 		this.sources = new ScriptSearch(this.jsSrcDir,this.sourceIncludes,this.sourceExcludes);
393 		this.specs = new ScriptSearch(this.jsTestSrcDir,this.specIncludes,this.specExcludes);
394 
395 		try {
396 			this.run();
397 		} catch(MojoFailureException e) {
398 			throw e;
399 		} catch(Exception e) {
400 			throw new MojoExecutionException("The jasmine-maven-plugin encountered an exception: \n"+this.stringifiesStackTraces.stringify(e),e);
401 		}
402 	}

452 	private void addRequireJsIfNecessary() {
453 		String scriptLoaderPath = this.getScriptLoaderPath() == null ? "require.js" : this.getScriptLoaderPath();
454 		String requireJsPath = String.format("%s/%s", this.jsSrcDir, scriptLoaderPath);
455 		if (SpecRunnerTemplate.REQUIRE_JS.equals(this.specRunnerTemplate)) {
456 			File requireJsFile = new File(requireJsPath);
457 			if (requireJsFile.exists()) {
458 				if (this.preloadSources == null) {
459 					this.preloadSources = new ArrayList<String>();
460 				}
461 				this.preloadSources.add(requireJsPath);
462 			}
463 		}
464 	}

533 	private File getResourceAsFile(String parameter, String resourceLocation) throws MojoExecutionException {
534 		File file = null;
535 
536 		if (resourceLocation != null) {
537 			this.locator.addSearchPath( "url", "" );
538 			this.locator.addSearchPath( FileResourceLoader.ID, this.mavenProject.getFile().getParentFile().getAbsolutePath() );
539 
540 			ClassLoader origLoader = Thread.currentThread().getContextClassLoader();
541 			try {
542 				Thread.currentThread().setContextClassLoader(this.getClass().getClassLoader());
543 				try {
544 					file = this.locator.getResourceAsFile(resourceLocation);
545 				} catch (Exception e) {
546 					throw new MojoExecutionException(String.format(ERROR_FILE_DNE,parameter,resourceLocation));
547 				}
548 			}
549 			finally {
550 				Thread.currentThread().setContextClassLoader( origLoader );
551 			}
552 		}
553 		return file;
554 	}
