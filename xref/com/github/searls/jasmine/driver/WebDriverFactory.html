57      Class<? extends WebDriver> webDriverClass = getWebDriverClass();
58      try {

65    private WebDriver createCustomWebDriver() throws Exception {
66      Constructor<? extends WebDriver> constructor = getWebDriverConstructor();
67      return constructor.newInstance(getWebDriverConstructorArguments(constructor));
68    }

70    private Object[] getWebDriverConstructorArguments(Constructor<? extends WebDriver> constructor) {
71      if (constructor.getParameterTypes().length == 0) {
72        return new Object[0];
73      } else
74        return new Object[] {getCapabilities()};
75    }

77    private Capabilities getCapabilities() {
78      DesiredCapabilities capabilities = new DesiredCapabilities();
79      capabilities.setJavascriptEnabled(true);
80      for (Map.Entry<String, String> entry : webDriverCapabilities.entrySet()) {
81        capabilities.setCapability(entry.getKey(), entry.getValue());
82      }
83      return capabilities;
84    }
