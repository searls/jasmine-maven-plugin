526 	private void addRequireJsIfNecessary() {
527 		String scriptLoaderPath = this.getScriptLoaderPath() == null ? "require.js" : this.getScriptLoaderPath();
528 		String requireJsPath = String.format("%s/%s", this.jsSrcDir, scriptLoaderPath);
529 		if (SpecRunnerTemplate.REQUIRE_JS.equals(this.specRunnerTemplate)) {
530 			File requireJsFile = new File(requireJsPath);
531 			if (requireJsFile.exists()) {
532 				if (this.preloadSources == null) {
533 					this.preloadSources = new ArrayList<String>();
534 				}
535 				this.preloadSources.add(requireJsPath);
536 			}
537 		}

593 	private File getResourceAsFile(String parameter, String resourceLocation) throws MojoExecutionException {
594 		File file = null;
595 
596 		if (resourceLocation != null) {
597 			this.locator.addSearchPath( "url", "" );
598 			this.locator.addSearchPath( FileResourceLoader.ID, this.mavenProject.getFile().getParentFile().getAbsolutePath() );
599 
600 			ClassLoader origLoader = Thread.currentThread().getContextClassLoader();
601 			try {
602 				Thread.currentThread().setContextClassLoader(this.getClass().getClassLoader());
603 				try {
604 					file = this.locator.getResourceAsFile(resourceLocation);
605 				} catch (Exception e) {
606 					throw new MojoExecutionException(String.format(ERROR_FILE_DNE,parameter,resourceLocation));
607 				}
608 			}
609 			finally {
610 				Thread.currentThread().setContextClassLoader( origLoader );
611 			}
612 		}
613 		return file;
