22  @RunWith(MockitoJUnitRunner.class)
23  public class ResourceRetrieverTest {
24  
25    @Mock
26    ResourceManager locator;
27  
28    @Mock
29    MavenProject mavenProject;
30  
31    ResourceRetriever subject;
32  
33    @Before
34    public void setUp() throws Exception {
35      subject = new ResourceRetriever(locator);
36      File mavenFile = mock(File.class);
37      when(mavenProject.getFile()).thenReturn(mavenFile);
38      File mavenDir = mock(File.class);
39      when(mavenFile.getParentFile()).thenReturn(mavenDir);
40      when(mavenDir.getAbsolutePath()).thenReturn("/maven/dir");
41    }
42  
43    @Test
44    public void testGetResource() throws ResourceNotFoundException, MojoExecutionException, MojoFailureException, FileResourceCreationException {
45      File expectedFile = mock(File.class);
46      String resourceLocation = "/foo/bar";
47      when(locator.getResourceAsFile(resourceLocation)).thenReturn(expectedFile);
48  
49      File actualFile = subject.getResourceAsFile("param", resourceLocation, mavenProject);
50  
51      assertThat(actualFile, is(expectedFile));
52    }
53  
54    @Test(expected = MojoExecutionException.class)
55    public void testResourceNotFound() throws ResourceNotFoundException, MojoExecutionException, MojoFailureException, FileResourceCreationException {
56      String resourceLocation = "/foo/bar";
57      when(locator.getResourceAsFile(resourceLocation)).thenThrow(new FileResourceCreationException("Bad Things Happen"));
58  
59      subject.getResourceAsFile("param", resourceLocation, mavenProject);
60    }
61  }
