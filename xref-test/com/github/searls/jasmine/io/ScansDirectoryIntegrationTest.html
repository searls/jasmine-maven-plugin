169   private List<String> slashify(String...scripts){
170     List<String> slashed = new ArrayList<String>();
171     for (String s : scripts) {
172       slashed.add(s.replace('/', File.separatorChar));
173     }
174     return slashed;
175   }
176 
177   private void createFile(String... paths) {
178     for (String path : paths) {
179       try {
180         new File(directory,path).createNewFile();
181       } catch (IOException e) {
182         throw new RuntimeException(e);
183       }
184     }
185   }
186 
187   private void createSubDir(String... names) {
188     for (String name : names) {
189       new File(directory,name).mkdir();
190     }
191   }
