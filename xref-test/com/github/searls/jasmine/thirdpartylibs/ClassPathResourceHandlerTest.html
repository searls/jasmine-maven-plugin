15  import static com.github.searls.jasmine.thirdpartylibs.ProjectClassLoaderHelper.projectClassLoaderOf;
16  import static org.mockito.Mockito.*;
17  
18  @RunWith(PowerMockRunner.class)
19  public class ClassPathResourceHandlerTest {
20    @Mock
21    private Request baseRequest;
22    @Mock
23    private HttpServletRequest request;
24    @Mock
25    private HttpServletResponse response;
26    @Mock
27    private Resource resource;
28    @Mock
29    private PrintWriter writer;
30  
31    private ClassPathResourceHandler subject;
32  
33    @Before
34    public void before() {
35      String jquery = "src/test/resources/webjars/jquery-1.10.2.jar";
36      subject = new ClassPathResourceHandler(projectClassLoaderOf(jquery));
37    }
38  
39    @Test
40    public void whenTargetStartsWithClassPathThenRespondWithResourceContent() throws Exception {
41      // given
42      when(baseRequest.isHandled()).thenReturn(false);
43      when(response.getWriter()).thenReturn(writer);
44  
45      // when
46      subject.handle("/META-INF/resources/webjars/jquery/1.10.2/jquery.js", baseRequest, request, response);
47  
48      // then
49      verify(writer).write(contains("jQuery JavaScript Library v1.10.2"));
50      verify(baseRequest).setHandled(true);
51    }
52  
53    @Test
54    public void whenResourceIsMissingThenDoNotProcess() throws Exception {
55      // given
56      when(baseRequest.isHandled()).thenReturn(false);
57      when(response.getWriter()).thenReturn(writer);
58  
59      // when
60      subject.handle("/notExistingResource", baseRequest, request, response);
61  
62      // then
63      verify(writer, never()).write(anyString());
64      verify(baseRequest, never()).setHandled(true);
65    }
66  }
