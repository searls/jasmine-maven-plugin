13  import org.junit.Before;
14  import org.junit.Test;

16  public class CoffeeScriptIntegrationTest {
17  
18    private static final String COFFEE =
19      "describe \"HelloWorld\", ->\n" +
20      "  it \"should say hello\", ->\n" +
21      "    hello_world = new HelloWorld\n" +
22      "    expect(hello_world.greeting()).toBe \"Hello, World\"";
23  
24    private static final String JAVASCRIPT =
25      "(function() {\n\n" +
26      "  describe(\"HelloWorld\", function() {\n" +
27      "    return it(\"should say hello\", function() {\n" +
28      "      var hello_world;\n" +
29      "      hello_world = new HelloWorld;\n" +
30      "      return expect(hello_world.greeting()).toBe(\"Hello, World\");\n" +
31      "    });\n" +
32      "  });\n\n" +
33      "}).call(this);\n";
34  
35    private CoffeeScript subject;
36    
37    private Map<String,String> mockCache;
38    
39    @Before
40    public void before() throws Exception {
41    	subject = new CoffeeScript();
42    	mockCache = Collections.synchronizedMap(new WeakHashMap<String,String>());
43    	injectFakeCache(mockCache);
44    }
45  
46    @Test
47    public void itCompiles() throws IOException {
48      String result = subject.compile(COFFEE);
49  
50      assertThat(result,is(JAVASCRIPT));
51    }
52  
53    @Test
54    public void itReliesOnTheCache() throws IOException {
55      String expected = "win";
56      subject.compile(COFFEE);
57      mockCache.put(StringEscapeUtils.escapeEcmaScript(COFFEE), expected);
58  
59      String result = subject.compile(COFFEE);
60  
61      assertThat(result,is(expected));
62    }
63  
64    private void injectFakeCache(Map<String,String> cacheMap) throws Exception {
65      Field cache = subject.getClass().getDeclaredField("cache");
66      cache.setAccessible(true);
67      cache.set(subject, cacheMap);
68    }
69  
70  }
