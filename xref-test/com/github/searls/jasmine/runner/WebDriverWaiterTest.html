14  import static org.hamcrest.Matchers.allOf;
15  import static org.mockito.Matchers.argThat;
16  import static org.mockito.Matchers.contains;
17  import static org.mockito.Mockito.verify;
18  import static org.mockito.Mockito.when;
19  
20  @RunWith(MockitoJUnitRunner.class)
21  public class WebDriverWaiterTest {
22  
23    @Rule
24    public ExpectedException expectedException = ExpectedException.none();
25  
26    @Mock
27    RemoteWebDriver webDriver;
28    int timeout = 2;
29    @Mock
30    Log log;
31  
32    WebDriverWaiter subject;
33  
34    @Before
35    public void setUp() throws Exception {
36      subject = new WebDriverWaiter();
37    }
38  
39    @Test
40    public void itShouldWait() throws Exception {
41      boolean debug = false;
42      when(webDriver.executeScript(WebDriverWaiter.EXECUTION_FINISHED_SCRIPT)).thenReturn(true);
43  
44      subject.waitForRunnerToFinish(webDriver, timeout, debug, log);
45    }
46  
47    @Test
48    public void itShouldThrowWhenTimesOut() throws Exception {
49      expectedException.expect(IllegalStateException.class);
50      expectedException.expectMessage("Timeout occurred.");
51  
52      boolean debug = false;
53      when(webDriver.executeScript(WebDriverWaiter.EXECUTION_FINISHED_SCRIPT)).thenReturn(false);
54  
55      subject.waitForRunnerToFinish(webDriver, timeout, debug, log);
56    }
57  
58    @Test
59    public void itShouldLogWhenTimesOutDebugging() throws Exception {
60      boolean debug = true;
61      when(webDriver.executeScript(WebDriverWaiter.EXECUTION_FINISHED_SCRIPT)).thenReturn(false);
62  
63      subject.waitForRunnerToFinish(webDriver, timeout, debug, log);
64  
65      verify(log).warn(argThat(allOf(new StringContains("Attempted to wait"), new StringContains(timeout + " seconds"))));
66      verify(log).warn(contains("Debug mode:"));
67    }
68  }
