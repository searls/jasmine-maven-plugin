52    @Before
53    public void setupGeneratorConfiguration() throws IOException{
54      when(this.generatorConfiguration.getSourceEncoding()).thenReturn(SOURCE_ENCODING);
55      when(this.generatorConfiguration.getReporterType()).thenReturn(ReporterType.HtmlReporter);
56      when(this.generatorConfiguration.getScriptResolver()).thenReturn(this.scriptResolver);
57      when(this.generatorConfiguration.getRunnerTemplate()).thenReturn(new IOUtilsWrapper().toString(SpecRunnerTemplate.DEFAULT.getTemplate()));
58      this.subject = new DefaultSpecRunnerHtmlGenerator(this.generatorConfiguration);
59    }
60  
61    @Test
62    public void shouldBuildBasicHtmlWhenNoDependenciesAreProvided() {
63      String html = this.subject.generate();
64  
65      assertThat(html, containsString("<html>"));
66      assertThat(html, containsString("</html>"));
67    }
68  
69    @Test
70    public void shouldPutInADocTypeWhenNoDependenciesAreProvided() throws Exception {
71      String html = this.subject.generate();
72  
73      assertThat(html, containsString(HTML5_DOCTYPE));
74      assertThat(this.getPage(html).getDoctype().getName(), is("html"));
75    }
76  
77    @Test
78    public void shouldAssignSpecifiedSourceEncoding() throws Exception {
79      String html = this.subject.generate();
80  
81      HtmlMeta contentType = this.getPage(html).getFirstByXPath("//meta");
82      assertThat(contentType.getContentAttribute(), is("text/html; charset=" + SOURCE_ENCODING));
83    }
84  
85    @Test
86    public void shouldDefaultSourceEncodingWhenUnspecified() throws Exception {
87      when(this.generatorConfiguration.getSourceEncoding()).thenReturn(null);
88  
89      String html = this.subject.generate();
90  
91      HtmlMeta contentType = this.getPage(html).getFirstByXPath("//meta");
92      assertThat(contentType.getContentAttribute(), is("text/html; charset=" + SpecRunnerHtmlGenerator.DEFAULT_SOURCE_ENCODING));
93    }

126   private HtmlPage getPage(String html) throws Exception {
127     MockWebConnection webConnection = new MockWebConnection();
128     webConnection.setDefaultResponse(html);
129     WebClient webClient = new WebClient();
130     webClient.setWebConnection(webConnection);
131     webClient.getOptions().setThrowExceptionOnScriptError(false);
132     webClient.setIncorrectnessListener(new IncorrectnessListener() {
133       @Override
134       public void notify(String arg0, Object arg1) {
135       }
136     });
137     return webClient.getPage("http://blah");
138   }
